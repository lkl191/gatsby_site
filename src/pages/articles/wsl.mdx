<H2title
    props='import'
/>

<Code
    copycode='wsl --import [distribution_name] [path/to/install/dir] [file_name].tar --version 2'
    langage='Shell'
/>


<H2title
    props='デフォルトユーザーの変更'
/>

<Code
    copycode='Function WSL-SetDefaultUser ($distro, $user) { Get-ItemProperty Registry::HKEY_CURRENT_USER\Software\Microsoft\Windows\CurrentVersion\Lxss\*\ DistributionName | Where-Object -Property DistributionName -eq $distro | Set-ItemProperty -Name DefaultUid -Value ((wsl -d $distro -u $user -e id -u) | Out-String); };'
/>

<Code
    copycode='WSL-SetDefaultUser <DistroName> <UserName>'
/>


<H2title
    props='export'
/>

<Code
    copycode='wsl --export [distribution_name] [file_name].tar'
/>
